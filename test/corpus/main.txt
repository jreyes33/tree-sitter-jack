Static and instance variables
====================================

class Dog {
  field int age, teeth;
  static boolean builtOnce;

  constructor Dog new(int ageParam, int teethParam) {
    let age = ageParam;
    let teeth = teethParam;
    let builtOnce = true;
    return this;
  }
}
